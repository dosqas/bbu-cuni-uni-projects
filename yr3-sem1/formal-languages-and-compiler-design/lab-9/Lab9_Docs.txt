[Problem 1]
### Redundant filtering
### INPUT
dataset cities = [
  {name: "Falticeni", population: 30000},
  {name: "Ludus", population: 17000},
  {name: "Boholt", population: 5000},
  {name: "Tandarei", population: 32000}
]

filter cities where population > 20000 and population > 3000
sort cities desc
select cities name
output cities



### OUTPUT
dataset cities = [
  {name: "Falticeni", population: 30000},
  {name: "Ludus", population: 17000},
  {name: "Boholt", population: 5000},
  {name: "Tandarei", population: 32000}
]

filter cities where population > 20000
sort cities desc
select cities name
output cities

              | <input> 
              | <filter> 
              | <select> 
              | <count> 
              | <group> 
              | <sort> 
              | <output> 
              | <if> 
              | <for>

              | <condition> <logical_op> <condition>

<nonzero_digit> ::= "1" | "2" | "3" | "4" | "5" | "6" | "7" | "8" | "9"

<letter> ::= "a" | "b" | ... | "z" | "A" | "B" | ... | "Z"
